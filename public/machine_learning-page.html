import pandas as pd
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.metrics.pairwise import cosine_similarity

class MLChatbot:
    def __init__(self, csv_file: str):
        data = pd.read_csv(csv_file)
        self.question_list = data["question"].tolist()
        self.answer_list = data["answer"].tolist()

        self.vectorizer = CountVectorizer()
        self.X = self.vectorizer.fit_transform(self.question_list)

    def reply(self, user_question: str) -> str:
        user_vec = self.vectorizer.transform([user_question])
        similarity = cosine_similarity(user_vec, self.X)
        idx = similarity.argmax()
        if similarity[0, idx] < 0.1:
            return "Sorry, I don't know the answer"
        return self.answer_list[idx]
